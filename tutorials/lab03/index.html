library(tidyr)
# Create a data frame
yawn_expt <- data_frame(group = c(rep("treatment", 34), rep("control", 16)),

# Make a contigency table with pipe %>%
yawn_expt %>%
  group_by(group, yawn) %>% 
  tally() %>%
  spread(yawn, n) %>% 
  group_by(group) %>%
  mutate(total = sum(no, yes))
